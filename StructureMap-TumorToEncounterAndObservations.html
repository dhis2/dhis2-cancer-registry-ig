Transform: Tumor Stage to CRTumorEncounter + Observations

map "https://dhis2.org/fhir/cancer-registry/StructureMap/TumorToEncounterAndObservations" = "TumorToEncounterAndObservations"

// Transform: Tumor Stage to CRTumorEncounter + Observations

uses "https://dhis2.org/fhir/cancer-registry/StructureDefinition/Tumor" alias Tumor as source
uses "http://hl7.org/fhir/StructureDefinition/Bundle" alias Bundle as target

group TumorToBundle(source src : Tumor, target tgt : Bundle) {
  src -> tgt.type = 'transaction' "setbundletype";
  // Create the Encounter resource
  src ->  tgt.entry as entry,  entry.resource = create('Encounter') as enc then {
    src -> enc.meta as meta then {
      src -> meta.profile = 'https://dhis2.org/fhir/cancer-registry/StructureDefinition/cr-tumor-encounter' "setencprofile";
    } "setencmeta";
    src -> enc.status = 'finished' "setencstatus";
    src -> enc.class as cls then {
      src -> cls.system = 'http://terminology.hl7.org/CodeSystem/v3-ActCode' "setclssystem";
      src -> cls.code = 'AMB' "setclscode";
    } "setencclass";
    src -> enc.type as tp then {
      src -> tp.coding as coding then {
        src -> coding.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-encounter-type' "settypesystem";
        src -> coding.code = 'tumor' "settypecode";
      } "settypecoding";
    } "setenctype";
    src.crIncidenceDate as dt ->  enc.period as p,  p.start = dt "setencperiod";
  } "createencounter";
  // Topography Observation
  src.crTopography as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "topostatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "topocodesys";
      src -> cd.code = 'topography' "topocodeval";
    } "topocode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRTopographyCS' "topovalsys";
      val -> cd.code = val "topovalcode";
    } "topovalue";
  } "createtopography";
  // Morphology Observation
  src.crMorphology as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "morphstatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "morphcodesys";
      src -> cd.code = 'morphology' "morphcodeval";
    } "morphcode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRMorphology4CS' "morphvalsys";
      val -> cd.code = val "morphvalcode";
    } "morphvalue";
  } "createmorphology";
  // Behaviour Observation
  src.crBehaviour as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "behavstatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "behavcodesys";
      src -> cd.code = 'behaviour' "behavcodeval";
    } "behavcode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRBehaviourCS' "behavvalsys";
      val -> cd.code = val "behavvalcode";
    } "behavvalue";
  } "createbehaviour";
  // Grade Observation (LOINC code)
  src.crGrade as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "gradestatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'http://loinc.org' "gradecodesys";
      src -> cd.code = '33732-9' "gradecodeval";
    } "gradecode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRGradeCS' "gradevalsys";
      val -> cd.code = val "gradevalcode";
    } "gradevalue";
  } "creategrade";
  // Basis of Diagnosis Observation
  src.crBasisDiagnosis as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "basisstatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "basiscodesys";
      src -> cd.code = 'basis-of-diagnosis' "basiscodeval";
    } "basiscode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRBasisDiagnosisCS' "basisvalsys";
      val -> cd.code = val "basisvalcode";
    } "basisvalue";
  } "createbasis";
  // Site Observation
  src.crSite as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "sitestatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "sitecodesys";
      src -> cd.code = 'site' "sitecodeval";
    } "sitecode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRSiteCS' "sitevalsys";
      val -> cd.code = val "sitevalcode";
    } "sitevalue";
  } "createsite";
  // Incidence Date Observation
  src.crIncidenceDate as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "incdatestatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "incdatecodesys";
      src -> cd.code = 'incidence-date' "incdatecodeval";
    } "incdatecode";
    val ->  obs.value = create('dateTime') as dt,  dt.value = val "incdatevalue";
  } "createincidencedate";
  // Age at Incidence Observation
  src.crAge as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "agestatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "agecodesys";
      src -> cd.code = 'age-at-incidence' "agecodeval";
    } "agecode";
    val ->  obs.value = create('integer') as i,  i.value = val "agevalue";
  } "createage";
  // Tumor Number Observation
  src.crTumorNumber as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "tnstatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "tncodesys";
      src -> cd.code = 'tumor-number' "tncodeval";
    } "tncode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRTumorNumberCS' "tnvalsys";
      val -> cd.code = val "tnvalcode";
    } "tnvalue";
  } "createtumornumber";
  // Address Observation
  src.crAddress as val ->  tgt.entry as entry,  entry.resource = create('Observation') as obs then {
    src -> obs.status = 'final' "addrstatus";
    src ->  obs.code as c,  c.coding as cd then {
      src -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/cr-observation-codes' "addrcodesys";
      src -> cd.code = 'address' "addrcodeval";
    } "addrcode";
    val ->  obs.value = create('CodeableConcept') as cc,  cc.coding as cd then {
      val -> cd.system = 'https://dhis2.org/fhir/cancer-registry/CodeSystem/CRAddressCS' "addrvalsys";
      val -> cd.code = val "addrvalcode";
    } "addrvalue";
  } "createaddress";
}
